set tabstop=4               " Tab宽度为4空格

set shiftwidth=4            " 自动缩进宽度

set expandtab               " Tab转空格

syntax on                   " 语法高亮

set hlsearch                " 高亮搜索结果

set incsearch               " 实时搜索

set ignorecase              " 搜索忽略大小写

set smartcase               " 智能大小写匹配
